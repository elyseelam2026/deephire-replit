export function getPresetForRole(title: string): RolePreset | null {
  return ROLE_PRESETS.find(p => p.titleMatch.some(r => r.test(title))) || null;
}

export function buildNapFromPreset(preset: RolePreset): NAPCriterion[] {
  return preset.criteria.map(critId => {
    const tmpl = CRITERION_TEMPLATES.find(t => t.id === critId)!;
    return {
      id: crypto.randomUUID(),
      requirement: tmpl.label,
      priority: 'non-negotiable',
      minFulfillment: 70,
      evidenceGuidance: `Look for: ${tmpl.subComponents.map(s => s.label).join(', ')}`,
      weight: tmpl.defaultTotalWeight,
      subComponents: tmpl.subComponents.map(s => ({
        ...s,
        weight: s.defaultWeight,
      })),
    };
  });
}
