export async function runDeepSearch(filters) {
  // Parse filters from UI/AI (mimics screenshot: title, location, exp, schools, etc.)
  const { title, location, experienceYears, schools, industries, skills, languages, keywords } = filters;
  const numResults = filters.numResults || 50;

  // Step 1: SerpAPI â€” Advanced People Search (Boolean for Recruiter filters)
  const serpQuery = {
    keywords: `"${title}" AND (${skills ? skills.join(' OR ') : ''}) AND (${schools ? schools.join(' OR ') : ''}) AND (${industries ? industries.join(' OR ') : ''})`,
    location,
    experience_level: experienceYears ? (experienceYears < 5 ? 'Entry Level' : 'Mid-Senior Level') : null,
    language: languages ? languages.join(',') : null,
    num_results: numResults,
    // Add more: zip_code, network: 'S', education: schools
  };
  console.log('ðŸ•·ï¸ SerpAPI Query:', serpQuery);

  const serpResults = await searchLinkedInPeople(serpQuery);
  if (!serpResults.profiles || serpResults.profiles.length === 0) {
    throw new Error('No matchesâ€”broaden filters');
  }

  // Step 2: Bright Data â€” Deep Scrape Top N Profiles
  const topProfiles = serpResults.profiles.slice(0, 20).map(p => p.profileUrl);
  const scrapedProfiles = [];
  for (const url of topProfiles) {
    try {
      const profileData = await scrapeLinkedInProfile(url); // Returns { name, title, company, exp, education, skills, languages, ... }
      scrapedProfiles.push({
        ...profileData,
        linkedin_url: url,
        fitScore: calculateFit(profileData, filters), // AI scorer
        verified: true // From real scrape
      });
    } catch (err) {
      console.log(`âŒ Scrape failed for ${url}: ${err.message}`);
    }
  }

  // Step 3: Sort/Filter (e.g., by fitScore > 7)
  const qualified = scrapedProfiles.filter(p => p.fitScore >= 7).sort((a, b) => b.fitScore - a.fitScore);

  console.log(`âœ… DeepSearch Complete: ${qualified.length} qualified from ${topProfiles.length} scraped`);
  return qualified;
}

// Simple AI Fit Scorer (expand with your ai.ts)
function calculateFit(profile, filters) {
  let score = 0;
  if (profile.title?.toLowerCase().includes(filters.title.toLowerCase())) score += 3;
  if (profile.experienceYears <= filters.experienceYears) score += 3;
  if (filters.schools.some(s => profile.education?.includes(s))) score += 2;
  if (filters.skills.some(s => profile.skills?.includes(s))) score += 2;
  return Math.min(score, 10);
}

import { SentenceTransformer } from 'transformers'; // npm install @xenova/transformers
const model = new SentenceTransformer('all-MiniLM-L6-v2');

export async function scoreCandidateVsJD(candidate, jdStruct) {
  const jdText = `
    Role: ${jdStruct.role}
    Need: ${jdStruct.need.join(', ')}
    Pain: ${jdStruct.pain}
    Metrics: ${jdStruct.metrics.join(', ')}
  `;

  const candidateText = `
    ${candidate.title} at ${candidate.company}
    ${candidate.summary || ''}
    Skills: ${candidate.skills?.join(', ')}
  `;

  const [jdEmb, candEmb] = await Promise.all([
    model.encode(jdText),
    model.encode(candidateText)
  ]);

  const similarity = cosineSimilarity(jdEmb, candEmb);
  const keywordBonus = jdStruct.keywords.filter(k => 
    candidateText.toLowerCase().includes(k.toLowerCase())
  ).length * 0.5;

  return Math.min(10, (similarity * 8) + keywordBonus).toFixed(1);
}

function cosineSimilarity(a, b) {
  const dot = a.reduce((sum, val, i) => sum + val * b[i], 0);
  const magA = Math.sqrt(a.reduce((sum, val) => sum + val * val, 0));
  const magB = Math.sqrt(b.reduce((sum, val) => sum + val * val, 0));
  return dot / (magA * magB);
}
