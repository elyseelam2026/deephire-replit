from playwright.sync_api import sync_playwright

def scrape_eqt_offices():
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()
        page.goto("https://eqtgroup.com/about/offices", timeout=60000)

        # Wait for the list of office cities to load
        page.wait_for_selector("a.OfficeListItemstyles__Link-sc-19u4nvv-1")

        offices = page.query_selector_all("a.OfficeListItemstyles__Link-sc-19u4nvv-1")

        results = []
        for office in offices:
            city_name = office.inner_text().strip()
            href = office.get_attribute("href")

            # Open the city page
            city_page = browser.new_page()
            city_page.goto(f"https://eqtgroup.com{href}", timeout=60000)
            city_page.wait_for_selector("div.ContentBlockstyles__Content-sc-1ln3jjr-1")

            # Extract address text
            address = city_page.text_content("div.ContentBlockstyles__Content-sc-1ln3jjr-1").strip()
            results.append({"city": city_name, "address": address})
            city_page.close()

        browser.close()

        return results

if __name__ == "__main__":
    offices = scrape_eqt_offices()
    for o in offices:
        print(f"{o['city']}\n{o['address']}\n{'-'*40}")
